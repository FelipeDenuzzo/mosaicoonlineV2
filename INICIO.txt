🎯 PARÂMETROS PRINCIPAIS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
--reference     Imagem base (obrigatório)
--tiles         Pasta com tiles (obrigatório)
--tile-size     Tamanho do tile (padrão: 30px)
--max-uses      Limite de repetições (padrão: ilimitado)
--output        Arquivo de saída (padrão: mosaico_final.png)

💡 EXEMPLOS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
# Básico
./create_mosaic.sh reference_images/foto.jpg tiles/

# Com tamanho customizado (20px)
./create_mosaic.sh reference_images/foto.jpg tiles/ --tile-size 20

# Com limite de uso (máximo 2 vezes cada tile)
./create_mosaic.sh reference_images/foto.jpg tiles/ --max-uses 2

# Especificar saída
./create_mosaic.sh reference_images/foto.jpg tiles/ --output output/meu_mosaico.png

🔍 COMO FUNCIONA
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. Analisa a cor média da imagem base em regiões pequenas
2. Para cada região, encontra a tile com cor mais próxima
3. Coloca a tile no lugar correspondente
4. Combina tudo em um único mosaico final

💪 DICAS DE PERFORMANCE
